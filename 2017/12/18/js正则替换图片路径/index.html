js正则替换图片路径

发表于 2017-12-18

把图片的相对路径改成绝对路径

var content = 'thisis<img src="abc.jpg" width="50" /><img src="cde.jpg" />';
var originHost='http://www.baidu.com/';
content.replace(/<img [^>]*src=['"]([^'"]+)[^>]*>/gi, function (match, capture) {
     // console.log('capture='+capture);
     // console.log('match='+match);
    var tmp ='';
    var absoluteSrcRegx=/http:/gi.test(capture)||/https:/gi.test(capture)
    if (!absoluteSrcRegx) {
    //    console.log('===')
        tmp= match.replace(capture,originHost+capture);
        return tmp
    }else{
        return match
    }
})
